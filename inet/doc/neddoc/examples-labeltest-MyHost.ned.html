package inet.examples.labeltest;

import inet.applications.tcpapp.TCPSessionApp;
import inet.transport.tcp.TCP;
import inet.nodes.inet.NetworkLayer;

import inet.linklayer.ethernet.EthernetInterface;
import inet.linklayer.ethernet.EthernetInterface2;
import inet.networklayer.common.InterfaceTable;
import inet.networklayer.ipv4.RoutingTable;
import inet.base.NotificationBoard;
module MyHost
{
    @node();
    @labels(node,ethernet-node);
    @display("bgb=391,279;i=device/pc3");
    gates:
        inout ethg @labels(EtherFrame-conn);
    submodules:
        interfaceTable: InterfaceTable {
            @display("p=327,109");
        }
        routingTable: RoutingTable {
            @display("p=327,171");
        }
        notificationBoard: NotificationBoard {
            @display("p=327,44");
        }
        tcpSessionApp: TCPSessionApp {
            @display("p=58,37");
        }
        tcp: TCP {
            @display("p=190,89");
        }
        networkLayer: NetworkLayer {
            @display("p=58,158");
        }
        ethernetInterface2: EthernetInterface2 {
            @display("p=190,218");
        }
    connections allowunconnected:
        tcpSessionApp.tcpOut --> tcp.appIn++;
        tcp.appOut++ --> tcpSessionApp.tcpIn;
        tcp.ipOut --> networkLayer.tcpIn;
        networkLayer.tcpOut --> tcp.ipIn;
        networkLayer.ifOut++ --> ethernetInterface2.netwIn;
        ethernetInterface2.netwOut --> networkLayer.ifIn++;
        ethernetInterface2.phys <--> ethg;
}
